###  Exempel

```python
from aspose import pycore
from aspose.cells import Workbook
from aspose.cells.drawing import CustomXmlShape, MsoDrawingType

# Instantiating a Workbook object
workbook = Workbook()
shapes = workbook.worksheets[0].shapes
# add a CustomXmlShape
shape = shapes.add_shape(MsoDrawingType.CUSTOM_XML, 3, 0, 3, 0, 50, 150)
# Check if a shape is CustomXmlShape
customXmlShape = None
if shapes[0].mso_drawing_type == MsoDrawingType.CUSTOM_XML:
    # is CustomXmlShape
    customXmlShape = pycore.cast(CustomXmlShape, shapes[0])

```
